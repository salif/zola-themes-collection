#Img shortcode

{{ img(src="fields.jpg",
    alt="Brown and Green Mountain View Photo",
    low=600,
    high=1200,
    full=true,
    caption="Brown and Green Mountain View",
    author="Christian Heitz",
    link="https://www.pexels.com/photo/brown-and-green-mountain-view-photo-842711/") }}

{{ img(src="fields.jpg",
    alt="Brown and Green Mountain View Photo") }}

#Video shortcode

{{ video(src="fields3.mp4",
    poster="fields3-poster.jpg",
    width=600,
    type="mp4",
    caption="Drone footage of green farmland",
    author="James Owen",
    link="https://www.pexels.com/video/drone-footage-of-farmland-4237250/") }}

{{ video(src="fields3.mp4") }}

#Audio shortcode

{{ audio(src="jingles.mp3",
    caption="Music Jingles 01",
    author="Kenney.nl",
    link="https://kenney.nl/assets/music-jingles") }}

{{ audio(src="jingles.mp3") }}
